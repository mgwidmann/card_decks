def add_combination method = nil, &block
  @combinations ||= []
  if block_given?
    @combinations << block
  elsif method
    @combinations << method.to_sym
  end
end

def wild value = nil, *suits, &block
  @wilds ||= {}
  if block_given?
    @wilds[:blocks] ||= []
    @wilds[:blocks] << block
  else
    if value == :joker
      @wilds[:joker] = true
    else
      suits = self.suits.keys if suits.empty?
      @wilds[value] = suits
    end
  end
end
